89  void add(const Eigen::Ref<const Eigen::VectorXd>& x)
90  {
91  if (W == 0.)
92  {
93  W = 1.;
94  T = x;
95  S.setZero();
96  return;
97  }
98  W += 1.;
99  T += x;
100  double f = 1. / W / (W - 1.);
101  dX = W * x - T;
102  for (int r = 0; r < D; r++)
103  {
104  for (int c = 0; c < D; c++)
105  {
106  S(r, c) += f * dX(r) * dX(c);
107  }
108  }
109  }

116  void add(double& W_, const Eigen::Ref<const Eigen::VectorXd>& T_,
117  const Eigen::Ref<const Eigen::VectorXd>& S_)
118  {
119  if (W == 0.)
120  {
121  W = W_;
122  T = T_;
123  S = S_;
124  return;
125  }
126  dX = T_ / W_ - T / W;
127 
128  double f = W * W_ / (W + W_);
129  for (int r = 0; r < D; r++)
130  {
131  for (int c = 0; c < D; c++)
132  {
133  S(r, c) += S_(r, c) + f * dX(r) * dX(c);
134  }
135  }
136  T += T_;
137  W += W_;
138  }
