238  double forwardTimeDistance(const Motion *a, const Motion *b) const
239  {
240  static const Eigen::VectorXd max_vel = si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->vel_limits_;
241 
242  const OMPLTimeIndexedRNStateSpace::StateType *sa = static_cast<const OMPLTimeIndexedRNStateSpace::StateType *>(a->state);
243  const OMPLTimeIndexedRNStateSpace::StateType *sb = static_cast<const OMPLTimeIndexedRNStateSpace::StateType *>(b->state);
244 
245  double ta, tb;
246  Eigen::VectorXd qa, qb;
247  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(a->state, qa, ta);
248  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(b->state, qb, tb);
249 
250  if (tb < ta) return 1e10;
251  Eigen::VectorXd diff = (qb - qa).cwiseAbs();
252  double min_dt = (diff.array() / max_vel.array()).maxCoeff();
253  if (fabs(tb - ta) < min_dt) return 1e10;
254  return si_->distance(a->state, b->state);
255  }

276  bool correctTime(const Motion *a, Motion *b, bool reverse, bool &changed) const
277  {
278  Eigen::VectorXd max_vel = si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->vel_limits_;
279  double ta, tb;
280  Eigen::VectorXd qa, qb;
281  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(a->state, qa, ta);
282  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(b->state, qb, tb);
283  Eigen::VectorXd diff = (qb - qa).cwiseAbs();
284  double min_dt = (diff.array() / max_vel.array()).maxCoeff();
285  if (fabs(tb - ta) < min_dt)
286  {
287  if (reverse_check_) return false;
288  tb = ta + (reverse ? -min_dt : min_dt);
289  changed = true;
290  }
291  else
292  changed = false;
293  if (tb < si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->tStart || tb > si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->tGoal) return false;
294  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->ExoticaToOMPLState(qb, tb, b->state);
295  return true;
296  }
